--- /etc/nginx/sites-enabled/Beyondmebtw ---

server {
    server_name beyondmebtw.com www.beyondmebtw.com;

    root  /bmbsifi/Beyondmebtw;
    index index.html;

    # Block unwanted bots
    if ($http_user_agent ~* (AhrefsBot|SemrushBot|MJ12bot|DotBot|BLEXBot|crawler)) {
        return 403;
    }

    # ── CORS ──────────────────────────────────────────────────────────────────
    # Wildcard is correct for public, credential-less static content AND it is
    # cache-safe: one cached response is valid for EVERY origin. A copy cached
    # during a same-origin visit here can therefore be reused cross-origin by
    # manage.beyondmebtw.com without a mismatch.
    #
    # The old config reflected the caller's origin ($cors_origin = $http_origin)
    # WITHOUT sending `Vary: Origin`. Caches key on URL only, so a no-CORS copy
    # cached during a same-origin visit got served to the cross-origin manage
    # page -> intermittent CORS errors (worked in incognito only because its
    # cache was cold). Switching to `*` removes the footgun entirely.
    add_header Access-Control-Allow-Origin  "*" always;
    add_header Access-Control-Allow-Methods "GET, POST, OPTIONS" always;
    add_header Access-Control-Allow-Headers "Content-Type, Authorization" always;

    # Preflight (handled once, globally)
    if ($request_method = OPTIONS) {
        return 204;
    }

    # Project gallery: SPA-style fallback to its own index.
    location /projects/ {
        try_files $uri $uri/ /projects/index.html;
    }

    # GitHub webhook deploy (beyondgitman.js)
    location /bmbg {
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_pass http://localhost:6009;
    }

    # Everything else: homepage, /blog/*.json, /manage/*.json, /about, etc.
    # Inherits the server-level CORS headers above.
    location / {
        try_files $uri $uri/ =404;
    }

    listen 443 ssl;
    ssl_certificate     /etc/letsencrypt/live/beyondmebtw.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/beyondmebtw.com/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;
}

# HTTP -> HTTPS redirect
server {

    if ($host = beyondmebtw.com)     { return 301 https://$host$request_uri; }

    server_name beyondmebtw.com www.beyondmebtw.com;
    listen 80;
    return 404;
}
